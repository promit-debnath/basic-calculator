oparetor =input("choose your oparator:+ - * /")

a =float(input("num1:"))
b =float(input("num2:"))

if oparetor == "+":
    result = a + b

    print(f"the ans is {result}")
elif oparetor == "-":
    result = a - b

    print(f"the ans is {result}")
elif oparetor == "*":
    result = a * b

    print(f"the ans is {result}")
elif oparetor == "/":
    result = a / b

    print(f"the ans is {result}")
